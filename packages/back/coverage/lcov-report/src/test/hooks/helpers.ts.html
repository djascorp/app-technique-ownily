  wait: (ms: number): Promise<void> => {

  /**
   * It takes a username and password, verifies them, and then returns a MetaUser object
   * @param {Context} ctx - Context - The context of the request.
   * @param credentials - { username: string; password: string }
   * @returns The userMeta object is being returned.
   */
  getMetaUser: async (ctx: Context, credentials: { username: string; password: string }): Promise<MetaUser> => {
    const userMeta: MetaUser = await ctx.broker.call("meta-service.idp.account.verify", credentials);
 
    userMeta.extended = await ctx.broker.call("service.users.get", { id: userMeta.sub });
 
    const companyIds = [...(userMeta.extended?.ownedCompanies ?? []), ...(userMeta.extended?.grantedCompanies ?? [])];
 
    userMeta.companies = await ctx.broker.call("service.companies.listAll", {
      ids: [...companyIds],
    });
 
    userMeta.subscriptions = await ctx.broker.call("service.subscriptions.list", {
      companyIds: [...companyIds],
    });
 
    userMeta.permissions = await ctx.broker.call("service.permissions.list", {
      userId: userMeta.sub,
    });
 
    return userMeta;
  },
 
  /* Creating a mock account for testing purposes. */
  createMockAccount: async (
    ctx: Context,
    credentials: { username: string; password: string },
    accountingPeriod?: { firstYear?: boolean; taxRegime?: TaxRegime; startAt?: string; endAt?: string }
  ): Promise<MetaUser> => {
    const userAccount: UserAccountResponse = await ctx.broker.call("meta-controller.idp.register.create", credentials);
 
    let userMeta = await Helpers.getMetaUser(ctx, credentials);
 
    /* Updating the user with the new data. */
    const updateUser: Partial<User> & { userId: string } = {
      userId: userAccount.sub,
      firstName: "Ownily",
      lastName: "TEST",
      email: credentials.username,
      phone: "0600000000",
      optinMarket: false,
    };
    await ctx.broker.call("controller.users.update", updateUser, { meta: { user: userMeta } });
 
    /* Creating a new company. */
    const newCompany: NewCompany = {
      name: "Test company",
      address: { street: "00 rue test street", city: "test city", zip: "00000" },
    };
    const company: Company = await ctx.broker.call(
      "controller.companies.create",
      { company: newCompany },
      { meta: { user: userMeta } }
    );
 
    // Wait finish emit event
    await Helpers.wait(2000);
 
    userMeta = await Helpers.getMetaUser(ctx, credentials);
 
    /* Creating an accounting period for the company. */
    const newAccountingPeriod: NewAccountingPeriod = {
      companyId: company.id,
      startAt: accountingPeriod?.startAt ? accountingPeriod.startAt : `${new Date().getFullYear()}-01-01`,
      endAt: accountingPeriod?.endAt ? accountingPeriod.endAt : `${new Date().getFullYear()}-12-31`,
      firstYear: accountingPeriod?.firstYear ? accountingPeriod.firstYear : false,
      taxRegime: accountingPeriod?.taxRegime ? accountingPeriod.taxRegime : TaxRegime.IR_2072,
    };
    await ctx.broker.call(
      "controller.companies.accounting-periods.create",
      { companyId: company.id, accountingPeriod: newAccountingPeriod },
      { meta: { user: userMeta } }
    );
 
    return await Helpers.getMetaUser(ctx, credentials);
  },
};
