import { IncomingMessage } from "connect";
import { Context, Service, ServiceBroker } from "moleculer";
import GlobalRoute from "./global.route";
 
/**
 * Expose route to /meta path
 */
export default class MetaRoute extends Service {
  public constructor(broker: ServiceBroker) {
    super(broker);
 
    this.parseServiceSchema({
      name: "route.meta",
      mixins: [new GlobalRoute(broker).originalSchema],
 
      // More info about settings: https://moleculer.services/docs/0.14/moleculer-web.html
      settings: {
        routes: [
          {
            name: "meta-route",
 
            // First Route
            // The mixins logic merge the settings of GlobalRoute, so first route will be merged with the default route.
            path: "/meta/v1",
 
            // Only 'name: controller.**' services will be valid for this route
            whitelist: ["meta-controller.idp.**"],
 
            aliases: {
              /** == Identity Provider ==**/
              // Account
              "GET /accounts/:username": "meta-controller.idp.account.get",
              "PUT /accounts/:username": "meta-controller.idp.account.update",
              "DELETE /accounts/:username": "meta-controller.idp.account.delete",
              "GET /accounts": "meta-controller.idp.account.list",
              // Register => All accounts with SUB
              "DELETE /registers/:sub": "meta-controller.idp.register.delete",
              // Retrieve JWT information
              "GET /login/me": "meta-controller.idp.login.getMe",
              // Alias
              "POST /renew": "meta-controller.idp.renew",
              "POST /impersonate/:sub": "meta-controller.idp.impersonate",
              "POST /depersonate": "meta-controller.idp.depersonate",
            },
          },
          {
            name: "meta-route-public",
 
            // Second route is for Public API
            // Mixin is not merger with global route
            // https://github.com/moleculerjs/moleculer-web/issues/101
            path: "/meta/v1",
 
            // Only 'name: controller.**' services will be valid for this route
            whitelist: ["meta-controller.idp.**"],
 
            authentication: false,
 
            bodyParsers: {
              json: true,
            },
 
            onBeforeCall(
              ctx: Context<Record<string, unknown>, { device: string | unknown }>,
              route: Record<string, unknown>,
              req: IncomingMessage & { method: string; url: string }
            ) {
              // We save Header, we need to retrieve user-agent in login Action
              // https://github.com/moleculerjs/moleculer-web/issues/117
              ctx.meta.device = req.headers["user-agent"];
 
              // Configure APM
              if (ctx.broker.$apm) {
                ctx.broker.$apm.setTransactionName(req.method + " " + req.url);
              }
            },
 
            // Public route for anonymous role
            aliases: {
              // Alias to register an account for a Member scope
              "POST /registers": "meta-controller.idp.register.create",
              // Alias to login or logout a user and manage cookie/JWT
              "POST /logout": "meta-controller.idp.logout",
              "POST /login": "meta-controller.idp.account.verify",
              "POST /accounts/update-password": "meta-controller.idp.account.update-password",
              "POST /accounts/reset-password": "meta-controller.idp.account.reset-password",
            },
          },
        ], // End routes
      },
    });
  }
}
